Recommended

Basic Integration

Simplest way to add the chatbot to your website with automatic initialization.

<!-- Add before closing </body> tag -->
<script 
  src="https://your-cdn.com/chatbot.js"
  data-auto-init="true"
  data-api-base-url="https://api.yoursite.com/api/v1"
></script>
Customized

With Configuration

Add custom configuration before loading the widget script.

<script>
  window.VetChatbotConfig = {
    apiBaseUrl: 'https://api.yoursite.com/api/v1',
    position: 'bottom-right',
    title: 'Pet Health Assistant',
    autoInit: true,
    context: {
      userName: 'John Doe',
      petName: 'Buddy',
      source: 'website'
    }
  };
</script>
<script src="chatbot.js"></script>
Advanced

Manual Initialization

Full control over when and how the widget is initialized.

<script src="chatbot.js"></script>
<script>
  // Initialize when ready
  const widget = VetChatbot.init({
    apiBaseUrl: 'https://api.yoursite.com/api/v1',
    position: 'bottom-left',
    title: 'Vet Assistant'
  });

  // Control methods
  // widget.destroy();
  // widget.reinit(newConfig);
</script>
No JavaScript

Data Attributes

Configure entirely through HTML data attributes.

<script
  src="chatbot.js"
  data-auto-init="true"
  data-api-base-url="https://api.yoursite.com/api/v1"
  data-position="bottom-right"
  data-title="Chat with us"
  data-user-name="Guest"
  data-pet-name="My Pet"
></script>

JavaScript API Reference

// Initialize with options
const widget = VetChatbot.init({
  apiBaseUrl: 'string',      // Required: Backend API URL
  position: 'bottom-right',  // 'bottom-right' | 'bottom-left'
  title: 'Vet Assistant',    // Widget header title
  containerId: 'vet-chatbot-root', // Custom container ID
  enabled: true,             // Enable/disable widget
  context: {                  // User context
    userId: 'string',
    userName: 'string',
    petName: 'string',
    source: 'string'
  }
});

// Instance methods
widget.destroy();           // Remove widget from DOM
widget.isInitialized();     // Check if widget is active
widget.getConfig();         // Get current configuration
widget.reinit(newConfig);   // Reinitialize with new config

// Static methods
VetChatbot.init(config);    // Initialize widget
VetChatbot.destroy();       // Destroy current instance
VetChatbot.isReady();       // Check if ready
VetChatbot.version;         // Get version string
